/* =============================================
   小飞数学练习卷生成器
   支持：10以内 / 20以内 / 100以内
   格式：横式 + 标准竖式
   ============================================= */

const MAX_NUM = 20;         // 最大数（改为 10 就是10以内）
const TOTAL = 20;            // 总题数（会被均分为横式和竖式）
const OPS = ['+', '-'];     // 运算符号，删掉 '-' 就只有加法

function randInt(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
}

function shuffle(arr) {
    const a = [...arr];
    for (let i = a.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [a[i], a[j]] = [a[j], a[i]];
    }
    return a;
}

function genProblem() {
    const op = OPS[randInt(0, OPS.length - 1)];
    let a, b, ans;
    if (op === '+') {
        a = randInt(1, MAX_NUM - 1);
        b = randInt(1, MAX_NUM - a);
        ans = a + b;
    } else {
        a = randInt(10, MAX_NUM);
        b = randInt(1, a);
        ans = a - b;
    }
    return { a, b, op, ans };
}

// 生成全部题目
const raw = [];
for (let i = 0; i < TOTAL; i++) {
    raw.push(genProblem());
}

// 随机打乱
const probs = shuffle(raw);

// 分配：前一半横式，后一半竖式
const half = Math.floor(TOTAL / 2);
const hProbs = probs.slice(0, half);   // 横式
const vProbs = probs.slice(half);      // 竖式

// ====== 渲染竖式（一行3道） ======
function renderVRow(items) {
    // items: [{a, b, op, ans}, ...]  每行3道
    const parts = items.map(p => {
        const opSym = p.op === '+' ? '＋' : '－';
        const a = String(p.a);
        const b = String(p.b);
        const w = Math.max(a.length, b.length) + 2; // padding
        const line = '─'.repeat(w);
        return [
            `  ${a.padStart(w)}`,
            `${opSym} ${b.padStart(w - 2)}`,
            ` ${line}`
        ].join('\n');
    });

    // 合并三列
    const lines = ['', '', ''];
    parts.forEach((block, idx) => {
        block.split('\n').forEach((line, li) => {
            if (idx === 0) {
                lines[li] = '  ' + line;
            } else {
                lines[li] += '          ' + line;
            }
        });
    });

    return lines.slice(1).join('\n'); // 去掉开头的空行
}

// 分组（每3道一行）
function chunk(arr, size) {
    const out = [];
    for (let i = 0; i < arr.length; i += size) {
        out.push(arr.slice(i, i + size));
    }
    return out;
}

const vRows = chunk(vProbs, 3);

// ====== 构建 Markdown 输出 ======
let md = '';
md += '# 🌟 数学练习卷 🌟\n\n';
md += '> **日期**：' + (await tp.date.now('YYYY-MM-DD')) + '　　**姓名**：__________　　**得分**：__________\n\n';

// ---- 横式 ----
md += '## 🌈 横式题（共 ' + hProbs.length + ' 题）\n\n';
md += '| 题号 | 题目 | 答案 | 题号 | 题目 | 答案 |\n';
md += '|:----:|:-----:|:----:|:----:|:-----:|:----:|\n';

for (let i = 0; i < hProbs.length; i += 2) {
    const p1 = hProbs[i];
    const p2 = hProbs[i + 1];
    const t1 = p1.a + ' ' + p1.op + ' ' + p1.b + ' =';
    if (p2) {
        const t2 = p2.a + ' ' + p2.op + ' ' + p2.b + ' =';
        md += `| ${i + 1} | ${t1} | | | ${i + 2} | ${t2} | |\n`;
    } else {
        md += `| ${i + 1} | ${t1} | | | | | |\n`;
    }
}

// ---- 竖式 ----
md += '\n---\n\n';
md += '## 🌈 竖式题（共 ' + vProbs.length + ' 题）\n\n';
md += '> 标准格式：数字右对齐，`＋` / `－` 在左侧，横线下写答案\n\n';
md += '```text\n';

vRows.forEach((row, ri) => {
    const rowNum = ri * 3 + 1;
    // 题号行
    const labelLine = row.map((_, i) => {
        const n = rowNum + i;
        return '第' + n + '题';
    }).join('          ');

    // 竖式内容
    const blocks = row.map(p => {
        const opSym = p.op === '+' ? '＋' : '－';
        const a = String(p.a);
        const b = String(p.b);
        const w = Math.max(a.length, b.length) + 2;
        const line = '─'.repeat(w);
        return [
            ' '.repeat(1) + a.padStart(w),
            opSym + ' ' + b.padStart(w - 2),
            ' ' + line
        ];
    });

    const maxH = Math.max(...blocks.map(b => b.length));
    for (let li = 0; li < maxH; li++) {
        if (li === 0) {
            md += labelLine + '\n';
        }
        md += blocks.map(b => b[li] || '').join('          ') + '\n';
    }
    md += '\n';
});

md += '```\n';

// ---- 答案 ----
md += '\n---\n\n';
md += '## ✅ 答案\n\n';
md += '| 题号 | 横式答案 | 题号 | 竖式答案 |\n';
md += '|:----:|:--------:|:----:|:--------:|\n';

const allLabeled = [...hProbs.map((p, i) => ({ ...p, label: i + 1, type: '横' })),
                     ...vProbs.map((p, i) => ({ ...p, label: i + 1, type: '竖' }))];

for (let i = 0; i < allLabeled.length; i += 2) {
    const p1 = allLabeled[i];
    const p2 = allLabeled[i + 1];
    if (p2) {
        md += `| ${p1.label}（${p1.type}） | ${p1.ans} | ${p2.label}（${p2.type}） | ${p2.ans} |\n`;
    } else {
        md += `| ${p1.label}（${p1.type}） | ${p1.ans} | | |\n`;
    }
}

md += '\n---\n\n';
md += '*🌻 小飞的数学练习卷 · 加油！*\n';

tR = md;
%>
